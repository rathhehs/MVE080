data = pd.read_csv("Goal_data.csv")

categories_order = [
    "Q1-13", "Q2-13", "Q3-13", "Q4-13",
    "Q1-14", "Q2-14", "Q3-14", "Q4-14",
    "Q1-15", "Q2-15", "Q3-15"
]
data['categories'] = pd.Categorical(data['categories'], categories=categories_order, ordered=True)

data['proportion'] = data.groupby('categories')['value'].transform(lambda x: (x / x.sum()) * 100)
data['proportion'] = data['proportion'].round(0) 

color_mapping = {
    "Miss": "#C00000",
    "Meet": "#999999", 
    "Exceed": "#4D4D4D" 
}

miss_data = data[data['what'] == 'Miss']

# draw the plot
stacked_chart = (
    ggplot(data, aes(x='categories', y='proportion', fill='what')) +
    geom_bar(stat='identity', position='stack') +
    geom_text(x=6.656, y=7, label="12%", ha="left", color="#FFFFFF") +
    geom_text(x=7.656, y=10, label="15%", ha="left", color="#FFFFFF") +
    geom_text(x=8.656, y=15, label="20%", ha="left", color="#FFFFFF") +
    geom_text(x=9.656, y=28, label="33%", ha="left", color="#FFFFFF") +
    geom_text(x=10.656, y=37, label="42%", ha="left", color="#FFFFFF") +

    labs(

        x="",

# Create data
data = pd.DataFrame({
    'Item': ['Pre']*5 + ['Post']*5,
    'category': ['Bored', 'Not great', 'OK', 'Kind of interested', 'Excited']*2,
    'proportion': [11, 5, 40, 25, 19, 12, 6, 14, 30, 38]
})

# Set the 'Item' column as a categorical variable with a order
data['Item'] = pd.Categorical(data['Item'], categories=['Pre', 'Post'], ordered=True)
data['category'] = pd.Categorical(data['category'], categories=['Bored', 'Not great', 'OK', 'Kind of interested', 'Excited'], ordered=True)

# Create the plot with updated colormap
plot = (
    ggplot(data, aes(x='Item', y='proportion', fill='category')) 
    + geom_bar(stat='identity', position='stack', width=0.5) 
    + scale_fill_manual(values=['#FF6347', '#4682B4', '#32CD32', '#8A2BE2', '#FFA07A']) 
    + labs(x='Time', y='Percentage [%]', fill='Category', title='Students become more excited about doing science after the program.') 
    + theme_minimal()
    + theme(
        axis_text_x=element_text(rotation=0, ha='center'),  
        figure_size=(10, 6),  
        panel_grid=element_blank() 
    )  
)

I use Excel to draw. I set "All" "Men" and "Women" as primary headers, with "Applicants" and "Admitted" as secondary headers beneath each category.
